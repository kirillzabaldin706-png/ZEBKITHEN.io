function cardBrand(num) {
  const n = num.replace(/\s/g, '');
  if (n.startsWith('4'))           return 'VISA';
  if (/^(22[0-2][0-9]|5[1-5])/.test(n)) return 'MC';
  if (/^(2200|2201|2202|6762|6304)/.test(n)) return 'MIR';
  return 'CARD';
}

Безопасность

Реализованные меры

  • XSS-защита: все данные из localStorage проходят через esc() и escJs() перед вставкой в DOM
  • Хэширование пароля: Admin Panel хранит только SHA-256 хэш в коде
  • Маскирование карт: хранятся только последние 4 цифры, без CVV и полного номера
  • HTTPS: GitHub Pages обеспечивает HTTPS по умолчанию

Рекомендации для продакшена

  • Вынести меню и заказы на сервер с базой данных
  • Использовать сертифицированный PCI DSS платёжный шлюз
  • Добавить Content Security Policy (CSP) заголовок
  • Настроить rate limiting на форме бронирования
  • Подключить мониторинг ошибок (Sentry, LogRocket)
